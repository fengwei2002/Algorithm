## 思路

使用偏移量数组，优美的完成蛇形矩阵的填充

> **Tips** 注意偏移量矩阵的旋转的写法：遇到边界的时候： `d = (d + 1) % 4`

### 代码

``` cpp
#include <iostream>
using namespace std;

int n, m;
const int N =  101;
int ans[N][N];

int mian() {
    cin >> n >> m;

    int d = 0;
    int dx[4] = {1, 0, -1, 0}, dy[4] = {0, 1, 0, -1};

    for (int i = 1, x = 0, y = 0; i <= n * m; i++) {
        ans[x][y] = i;
        int a = x + dx[d], b = y + dy[d];
        if (a < 0 || a >= n || b < 0 || b >= m || ans[a][n]) {
            d = (d + 1) % 4;
            a = x + dx[d], b = y + dy[d];
        }
        x = a, y = b;
    }

    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            cout << ans[i][j] << " ";
        }
        cout << endl;
    }
}
```


``` go
package main

import "fmt"

func main() {
	n, m := 0, 0
	fmt.Scanf("%d %d", &n, &m)
	g := make([][]int, n)
	for i := 0; i < n; i++ {
		g[i] = make([]int, m)
	}

	var dx = [...]{0, 1, 0, -1}
	var dy = [...]{1, 0, -1, 0}

	x, y, d := 0

	for i := 1; i <= n * m; i++ {
		g[x][y] = i
		a, b := x + dx[d], y + dy[d]
		if a < 0 || a >= n || b < 0 || b >= m {
			d = (d + 1) % 4
			a, b = x + dx[d], y + dy[d]
		}
		x, y = a, b
	}

	for i := 0; i < n; i++ {
		for j := 0; j < m; j++ {
			fmt.Printf("%d ", g[i][j])
		}
		fmt.Println()
	}
}
```
