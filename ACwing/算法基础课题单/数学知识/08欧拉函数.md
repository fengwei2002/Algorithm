对于正整数 n, 欧拉函数是小于或等于 n 的正整数中与 n 互质的数的数目

``` cpp
 #include <iostream>
 using namespace std;

 int phi(int x) {
     int res = x;
     for (int i = 2; i <= x / i; i++) {
         if (x % i == 0) {
             res = res / i * (i - 1);
             while (x % i == 0) x /= i;
         }
         if (x > 1) res = res / x * (x - 1);
     }
     return res;
 }

 int main() {
     int n;
     cin >> n;
     while (n--) {
         int x;
         cin >> x;
         cout << phi(x) << endl;
     }
     return 0;
 }
```
