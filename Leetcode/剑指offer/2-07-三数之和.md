### C++

``` cpp
class Solution {
   public:
    vector<vector<int>> threeSum(vector<int>& nums) {
        vector<vector<int>> ans;

        int n = nums.size();
        sort(nums.begin(), nums.end()); // 记得要优先排序
 
        for (int i = 0; i < n; i++) {
            if (nums[i] > 0 || (i > 0 && nums[i] == nums[i - 1])) continue; // 这一步是去掉相同的 i

            int k = n - 1; // k 是逆序开始的一个指针
            int j = i + 1; // j 是 i 后面的一个指针

            while (j < k) {
                if (nums[i] + nums[j] + nums[k] > 0) {
                    k--;
                } else if (nums[i] + nums[j] + nums[k] < 0) {
                    j++;
                } else { // 找到了一个三元组
                    ans.push_back({nums[i], nums[j], nums[k]}); 

                    // 在找到结果后进行去重，而不是在找到结果前进行去重
                    while (k > j && nums[k] == nums[k - 1]) k--; // 这两步是对于一个 i 去掉相同的 j 和 k
                    while (k > j && nums[j] == nums[j + 1]) j++;
                    k--, j++;
                }
            }
        }
        return ans;
    }
};
```

### GO

``` go
func threeSum(nums []int) [][]int {
    ans := [][]int{}
    n := len(nums)  
    sort.Ints(nums)

    for i := 0; i < n; i++ {
        if nums[i] > 0 || (i > 0 && nums[i] == nums[i - 1]) {
            continue;
        }

        j, k := i + 1, n - 1

        for j < k {
            val := nums[i] + nums[j] + nums[k]
            if val > 0 {
                k--
            } else if val < 0 {
                j++
            } else if val == 0 {
                ansItem := []int{nums[i], nums[j], nums[k]}
                ans = append(ans, ansItem)

                for j < k && nums[j] == nums[j + 1] {
                    j++
                }
                for j < k && nums[k] == nums[k - 1] {
                    k--
                }
                j, k = j + 1, k - 1
            }
        }
    }
    return ans
}
```
