### C++

``` cpp
class Solution {
public:
    int singleNumber(vector<int>& nums) {
        int ans = 0;
        for (int i = 0; i < 32; i++) {
            int total = 0;
            for (int num: nums) {
                total += ((num >> i) & 1);
            }
            if (total % 3 == 1) {
                ans += (1 << i);
            }
        }
        return ans;
    }
};
```

### GO

``` go
func singleNumber(nums []int) int {
    ans := int32(0)
    for i := 0; i < 32; i++ {
        total := int32(0)
        for _, num := range nums {
            total += (int32(num) >> i ) & 1
        }
        if total % 3 == 1 {
            ans += (1 << i)
        }
    }
    return int(ans)
}
```
