### C++

``` cpp
class Solution {
public:
    int numSubarrayProductLessThanK(vector<int>& nums, int k) {
        int ans = 0, p = 1;
        for (int i = 0, j = 0; i < nums.size(); i++) {
            p *= nums[i];
            while (j <=i && p >= k) p /= nums[j++];
            ans += i - j + 1;
        }
        return ans;
    }
};
```

``` go
func numSubarrayProductLessThanK(nums []int, k int) int {
    ans, p := 0, 1
    for i, j := 0, 0; i < len(nums); i++ {
        p *= nums[i]
        for i >= j && p >= k {
            p /= nums[j]
            j++
        }
        ans += i - j + 1
    }
    return ans
}
```
