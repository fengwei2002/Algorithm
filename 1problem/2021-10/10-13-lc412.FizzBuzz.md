// https://github.com/fengwei2002/algorithm

class Solution {
public:
    vector<string> fizzBuzz(int n) {
        vector<string> ans(n, string(""));
        for (int i = 1; i <= n; i++) {
            if (i % 15 == 0) {
                ans[i - 1] = "FizzBuzz";
            } else if (i % 3 == 0) {
                ans[i - 1] = "Fizz";
            } else if (i % 5 == 0) {
                ans[i - 1] = "Buzz";
            } else {
                ans[i - 1] = to_string(i);
            }
        }
        return ans;
    }
};
