class Solution {
public:
    int findMinFibonacciNumbers(int k) {
        vector<int> f(0);
        if (k == 1) {
            return 1;
        }
        f.push_back(1);
        f.push_back(1);
        for (; f.back() <= k; ) {
            f.push_back(f.rbegin()[1] + f.back());
        }
        f.pop_back();
        
        int count = 0;
        int index = f.size() - 1;
        while (k > 0) {
            if (k >= f[index]) {
                k -= f[index];
                count++;
            } else {
                index--;
            }
        }

        return count;
    }
};

class Solution:
    def findMinFibonacciNumbers(self, k: int) -> int:
        if k == 1:
            return 1
        
        fib = [1, 1]
        while fib[-1] <= k:
            fib.append(fib[-1] + fib[-2])
        fib = fib[:-1]

        index = len(fib) - 1
        count = 0
        while k > 0:
            if k >= fib[index]:
                k -= fib[index]
                count += 1
            else:
                index -= 1
        
        return count

func findMinFibonacciNumbers(k int) int {
    if k == 1 {
        return 1
    }

    f := make([]int, 0)
    f = append(f, 1)
    f = append(f, 1)

    for f[len(f) - 1] <= k {
        f = append(f, f[len(f) - 2] + f[len(f) - 1])
    }

    index := len(f) - 1
    count := 0

    for k > 0 {
        if k >= f[index] {
            k -= f[index]
            count++
        } else {
            index--
        }
    }

    return count
}
