简单的使用 reverse 库函数：


``` CPP
class Solution {
public:
    string reversePrefix(string word, char ch) {
        int index = 0;
        for (int i = 0; i < word.size(); i++) {
            if (word.at(i) == ch) {
                index = i;
                break;
            }
        }
        reverse(word.begin(), word.begin() + index + 1);
        return word;
    }
};
```

手写 reverse：

``` cpp
class Solution {
public:
    string reversePrefix(string word, char ch) {
        int index = 0;
        for (int i = 0; i < word.size(); i++) {
            if (word.at(i) == ch) {
                index = i;
                break;
            }
        }
        int l = 0, r = index;
        while (l < r) {
            swap(word[l++], word[r--]);
        }
        return word;
    }
};
```
