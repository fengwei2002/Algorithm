## 思路


奶牛有两个斑点，农夫将两个斑点连成一个端点的最小补充面积

>  双端队列广搜可以在线性时间内完成

简单做法就是先找出两个连通块在哪
然后暴力枚举两个连通块中每一个点的曼哈顿距离
最短的距离就是最优解

## 代码

使用 DFS 进行搜索两个连通块

``` cpp 
#include <iostream>
#include <algorithm>

using namespace std;
const int N = 55;
typedef pair<int, int> PII;
#define x first
#define y second

using namespace std;

int n, m;
char g[N][N];
vector<PII> points[2];
// 把两个集合中的点存起来

int dx[4] = {-1, 0, 1, 0}, dy[4] = {0, 1, 0, -1};

void dfs(int x, int y, vector<PII> & ps) {
    g[x][y] = '.';
    ps.push_back({x, y});

    for (int i = 0; i < 4; i ++ ) {
        int a = x + dx[i], b = y + dy[i];
        // 枚举上下左右四个方向
        if (a >= 0 && a < n && b >= 0 && b < m && g[a][b] == 'X')
            dfs(a, b, ps);
    }
}

int main() {
    cin >> n >> m;
    for (int i = 0; i < n; i++) cin >> g[i];
    
    for (int i = 0, k = 0; i < n; i++)
        for (int j = 0; j < m; j++) 
            if (g[i][j] == 'X') 
                dfs(i, j, points[k++]);
                // 把和 i, j 相连的连通块存到 points k 里面
    
    // 这样的话就得到了两个连通块， 存储再 points 中
    int ans = 1e8;
    for (auto & a : points[0])
        for (auto & b : points[1])
            ans = min(ans, abs(a.x - b.x) + abs(a.y - b.y) - 1);
    
    cout << ans << endl;
    return 0;
}
```

使用 bfs 进行连通块的搜索

``` cpp 
#include <cstring>
#include <iostream>
#include <algorithm>
#include <queue>

using namespace std;

typedef pair<int, int> PII;

const int N = 55, INF = 0x3f3f3f3f;

int n, m, cnt;
char g[N][N];
bool st[N][N];

vector<PII> points[2];

void bfs(int x, int y, vector<PII> & pointArray) {
    queue<PII> q;
    q.push({x, y});
    st[x][y] = true;

    while (q.size()) {
        auto t = q.front();
        q.pop();
        pointArray.push_back(t);

        int tx = t.first, ty = t.second;
        int dx[4] = {-1, 0, 1, 0}, dy[4] = {0, 1, 0, -1};
        for (int i = 0; i < 4; i++) {
            int a = tx + dx[i], b = ty + dy[i];
            if (a >= 0 && a < n && b >= 0 && b < m && g[a][b] == 'X' && !st[a][b]) {
                q.push({a, b});
                st[a][b] = true;
            } else {
                continue;
            }
        }
    }
}

int main() {
    cin >> n >> m;
    for (int i = 0; i < n; i++) cin >> g[i];

    // 分别保存两个连通块中的所有点
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            if (g[i][j] == 'X' && !st[i][j]) {
                bfs(i, j, points[cnt++]);
            }
        }
    }
    
    int ans = INF;
    for (auto & a : points[0]) {
        for (auto & b : points[1]) {
            ans = min(ans, abs(a.first - b.first) 
                         + abs(a.second - b.second) - 1);
        }
    }

    cout << ans << endl;
    return 0;
}
```
